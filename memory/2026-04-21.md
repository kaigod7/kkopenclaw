def is_aedt(dt):  # 悉尼夏令时判断
    y, m = dt.year, dt.month
    if m == 10: return dt.day > first_sunday(y, 10) or (dt.day == first_sunday(y, 10) and dt.hour >= 1)
    elif m == 4: return dt.day < first_sunday(y, 4) or (dt.day == first_sunday(y, 4) and dt.hour < 2)
    elif 5 <= m <= 9: return False
    else: return True
